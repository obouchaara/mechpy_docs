E, nu = sp.symbols("E nu")
lamda, mu = sp.symbols("lamda mu", cls=sp.Function)
lamda = (E * nu) / ((1 + nu) * (1 - 2 * nu))
mu = E / (2 * (1 + nu))
display(lamda, mu)
E, nu = sp.symbols("E nu", cls=sp.Function)
E = mu * (3 * lamda + 2 * mu) / (lamda + mu)
nu = lamda / (2 * (lamda + mu))
display(E, nu)
display(sp.simplify((E)), sp.simplify(nu))
